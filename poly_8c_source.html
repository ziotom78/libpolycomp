145 static double integer_power(int x, int y)
146 {
147  double dbl_x = (double)x;
148 
149  if (y < 0)
150  abort();
151 
152  if (y == 0)
153  return 1.0;
154  else if (y == 1)
155  return dbl_x;
156  else {
157  double result = dbl_x * dbl_x;
158  int cur_power = 2;
159  while (2 * cur_power < y) {
160  result *= result;
161  cur_power *= 2;
162  }
163 
164  if (y > cur_power)
165  result *= integer_power(x, y - cur_power);
166 
167  return result;
168  }
169 }

203 pcomp_poly_fit_data_t* pcomp_init_poly_fit(size_t num_of_samples,

211  poly_fit->num_of_coeffs = num_of_coeffs;
212  poly_fit->workspace
213  = gsl_multifit_linear_alloc(num_of_samples, num_of_coeffs);
214  poly_fit->matrix = gsl_matrix_alloc(num_of_samples, num_of_coeffs);
215  poly_fit->y = gsl_vector_alloc(num_of_samples);

219

221  for (j = 0; j < num_of_coeffs; ++j) {

224  }
225  }

328 int pcomp_run_poly_fit(pcomp_poly_fit_data_t* poly_fit, double* coeffs,
329  const double* points)
330 {
331  size_t idx;
332  double chisq;
333 
334  if (poly_fit == NULL || coeffs == NULL || points == NULL)
335  abort();
336 
337  for (idx = 0; idx < poly_fit->num_of_samples; ++idx) {
338  gsl_vector_set(poly_fit->y, idx, points[idx]);
339  }
340 
341  if (gsl_multifit_linear(poly_fit->matrix, poly_fit->y, poly_fit->c,
342  poly_fit->cov_matrix, &chisq,
343  poly_fit->workspace) != 0) {
344  return PCOMP_STAT_INVALID_FIT;
345  }
346 
347  for (idx = 0; idx < poly_fit->num_of_coeffs; ++idx) {
348  coeffs[idx] = gsl_vector_get(poly_fit->c, idx);
349  }
350 
351  return PCOMP_STAT_SUCCESS;
352 }

499 int pcomp_run_chebyshev(pcomp_chebyshev_t* plan,
500  pcomp_transform_direction_t dir, double* output,
501  const double* input)
502 {
503  double norm;
504  size_t idx;
505 
506  if (plan == NULL)
507  abort();
508 
509  if (input != NULL) {
510  for (idx = 0; idx < plan->num_of_samples; ++idx) {
511  plan->input[idx] = input[idx];
512  }
513  }
514 
515  fftw_execute(plan->fftw_plan_ptr);
516  norm = chebyshev_normalization(dir, plan->num_of_samples);
517 
518  for (idx = 0; idx < plan->num_of_samples; ++idx) {
519  plan->output[idx] *= norm;
520  }
521 
522  if (output != NULL && output != plan->output) {
523  for (idx = 0; idx < plan->num_of_samples; ++idx) {
524  output[idx] = plan->output[idx];
525  }
526  }
527 
528  return PCOMP_STAT_SUCCESS;
529 }

737 static double eval_poly(double* coeffs, size_t num_of_coeffs, double x)
738 {
739  if (coeffs == NULL)
740  abort();
741 
742  if (num_of_coeffs >= 1) {
743  size_t idx = num_of_coeffs - 1;
744  double result = coeffs[idx];
745 
746  if (num_of_coeffs == 1)
747  return result;
748 
749  --idx;
750  while (1) {
751  result = result * x + coeffs[idx];
752 
753  if (idx > 0)
754  --idx;
755  else
756  break;
757  }
758 
759  return result;
760  }
761  else
762  return 0.0;
763 }

790 void pcomp_straighten(double* output, const double* input,
791  size_t num_of_samples, double period)
792 {
793  size_t idx;
794 
795  if (input == NULL || output == NULL)
796  abort();
797 
798  if (period > 0) {
799  double half_period = period * 0.5;
800  double offset = 0.0;
801 
802  output[0] = input[0];
803 
804  for (idx = 1; idx < num_of_samples; ++idx) {
805  double diff_with_previous = input[idx] - input[idx - 1];
806  if (diff_with_previous > half_period)
807  offset -= period;
808  else if (diff_with_previous < -half_period)
809  offset += period;
810 
811  output[idx] = input[idx] + offset;
812  }
813  }
814  else {
815  for (idx = 0; idx < num_of_samples; ++idx)
816  output[idx] = input[idx];
817  }
818 }

968 size_t pcomp_chunk_num_of_bytes(const pcomp_polycomp_chunk_t* chunk)
969 {
970  if (chunk == NULL)
971  abort();
972 
973  if (chunk->is_compressed) {
974  /* The size is calculated as follows:
975  * - the "compressed" flag (int8_t)
976  * - the number of samples (pcomp_chunk_size_t)
977  * - the number N of polynomial coefficients (pcomp_poly_size_t)
978  * - the size of the Chebyshev mask
979  * - the number M of Chebyshev coefficients (pcomp_chunk_size_t)
980  * - Nx8 bytes for the polynomial
981  * - Mx8 bytes for the Chebyshev coefficients
982  */
983  return sizeof(int8_t) + sizeof(pcomp_chunk_size_t)
984  + sizeof(pcomp_poly_size_t)
985  + pcomp_chunk_cheby_mask_size(chunk->num_of_samples)
986  + sizeof(pcomp_chunk_size_t)
987  + (chunk->num_of_poly_coeffs
988  + chunk->num_of_cheby_coeffs) * sizeof(double);
989  }
990  else {
991  /* The size is calculated as follows:
992  * - 1 byte for the "uncompressed" flag
993  * - 4 bytes for the number of samples
994  * - Nx8 bytes for the samples
995  */
996  return sizeof(int8_t) + sizeof(pcomp_chunk_size_t)
997  + chunk->num_of_samples * sizeof(double);
998  }
999 }

1080 /***********************************************************************
1081  * This routine implements the core of the "polynomial compression": a
1082  * polynomial fitting plus a Chebyshev transform on the residuals. It
1083  * is a static function because we're going to wrap a nicer interface
1084  * around it.
1085  */
1086 
1087 static int polyfit_and_chebyshev(pcomp_polycomp_t* params,
1088  double* coeffs, const double* input,
1089  double* max_residual)
1090 {
1091  size_t idx;
1092  int status;
1093  double running_max = -1.0; /* Negative stands for "uninitialized" */
1094 
1095  status = pcomp_run_poly_fit(params->poly_fit, coeffs, input);
1096  if (status != PCOMP_STAT_SUCCESS)
1097  return status;
1098 
1099  for (idx = 0; idx < params->samples_per_chunk; ++idx) {
1100  double abs_residual;
1101 
1102  params->chebyshev->input[idx]
1103  = input[idx]
1104  - eval_poly(coeffs, params->poly_fit->num_of_coeffs,
1105  idx + 1.0);
1106

1108  if (abs_residual > running_max || running_max < 0.0)
1109  running_max = abs_residual;
1110  }
1111

1113  *max_residual = running_max;
1114 
1115  if (params->algorithm != PCOMP_ALG_NO_CHEBYSHEV) {
1116  status = pcomp_run_chebyshev(

1118  }
1119

1123 /***********************************************************************
1124  * Sort the array "positions" according to the absolute values of
1125  * "coeffs", in *descending* order. The function uses the merge sort
1126  * algorithm. */
1127 
1128 static void sort_positions(pcomp_chunk_size_t positions[],
1129  const double coeffs[], size_t num)
1130 {
1131  size_t front, back;
1132  double pivot;
1133  pcomp_chunk_size_t temp;
1134 
1135  if (num < 2)
1136  return;
1137 
1138  pivot = fabs(coeffs[positions[num / 2]]);
1139  for (front = 0, back = num - 1;; front++, back--) {
1140  while (fabs(coeffs[positions[front]]) > pivot) {
1141  front++;
1142  }
1143 
1144  while (pivot > fabs(coeffs[positions[back]])) {
1145  if (back == 0)
1146  break;
1147  back--;
1148  }
1149 
1150  if (front >= back)
1151  break;
1152 
1153  temp = positions[front];
1154  positions[front] = positions[back];
1155  positions[back] = temp;
1156  }
1157  sort_positions(positions, coeffs, front);
1158  sort_positions(positions + front, coeffs, num - front);
1159 }

1171 static double compute_discrepancy(double a[], double b[], size_t num)
1172 {
1173  size_t idx;
1174  double err = 0.0;
1175  for (idx = 0; idx < num; ++idx) {
1176  double cur_err = fabs(a[idx] - b[idx]);
1177  if (cur_err > err)
1178  err = cur_err;
1179  }

1184 /* On exit, the bits in "bitmask" will be set to 1 in correspondence of
1185  * every Chebyshev coefficient that must be retained. The function
1186  * returns the number of Chebyshev coefficients to retain (i.e., the
1187  * number of bits in "mask" that have been set to 1).*/
1188 static size_t trunc_chebyshev(pcomp_chebyshev_t* chebyshev,
1189  pcomp_chebyshev_t* inv_chebyshev,
1190  double max_allowable_error, uint8_t* mask,
1191  double* max_error)
1192 {
1193  size_t idx;
1194  size_t cur_coeff = 0;
1195  pcomp_chunk_size_t* positions;
1196  double err;
1197 
1198  if (chebyshev == NULL || inv_chebyshev == NULL || mask == NULL
1199  || chebyshev->num_of_samples != inv_chebyshev->num_of_samples)
1200  abort();
1201 
1202  if (max_allowable_error <= 0.0)
1203  return 0;
1204 
1205  /* The "positions" array contains the indexes to the
1206  * chebyshev->output array, i.e., the list of Chebyshev
1207  * coefficients to sort in decreasing order. At the beginning each
1208  * entry is set to its own index:
1209  *
1210  * +---+---+---+-----+
1211  * positions: | 0 | 1 | 2 | ... |
1212  * +---+---+---+-----+
1213  *
1214  * After the call to "sort_positions", the "positions" array
1215  * contains the indexes to "chebyshev->output" ordered according
1216  * to the decreasing absolute value of the latters, e.g.:
1217  *
1218  * +---+---+---+-----+
1219  * positions: | 7 | 2 | 5 | ... |
1220  * +---+---+---+-----+
1221  */
1222  positions = malloc(chebyshev->num_of_samples
1223  * sizeof(pcomp_chunk_size_t));
1224  for (idx = 0; idx < chebyshev->num_of_samples; ++idx) {
1225  positions[idx] = idx;
1226  }
1227  sort_positions(positions, chebyshev->output,
1228  chebyshev->num_of_samples);
1229 
1230  /* Start by setting all the coefficients to zero */
1231  memset(&inv_chebyshev->input[0], 0,
1232  chebyshev->num_of_samples * sizeof(inv_chebyshev->input[0]));
1233  memset(&mask[0], 0,
1234  pcomp_chunk_cheby_mask_size(chebyshev->num_of_samples));
1235 
1236  /* Add the coefficients one by one until the error is below
1237  * "max_allowable_error" */
1238  cur_coeff = 0;
1239  while (cur_coeff < chebyshev->num_of_samples) {
1240 
1241  inv_chebyshev->input[positions[cur_coeff]]
1242  = chebyshev->output[positions[cur_coeff]];
1243  set_bit(mask, positions[cur_coeff]);
1244  ++cur_coeff;
1245 
1246  pcomp_run_chebyshev(inv_chebyshev, inv_chebyshev->dir, NULL,
1247  NULL);
1248  err = compute_discrepancy(chebyshev->input,
1249  inv_chebyshev->output,
1250  chebyshev->num_of_samples);
1251 
1252  if (err < max_allowable_error)
1253  break;
1254  }
1255 
1256  if (max_error != NULL)
1257  *max_error = err;
1258 
1259  return cur_coeff;
1260 }

1281 int pcomp_run_polycomp_on_chunk(pcomp_polycomp_t* params,
1282  const double* input,
1283  pcomp_chunk_size_t num_of_samples,
1284  pcomp_polycomp_chunk_t* chunk,
1285  double* max_error)
1286 {
1287  uint8_t* mask = NULL;
1288  double* coeffs = NULL;
1289  size_t cheby_coeffs_to_retain = 0;
1290  int apply_chebyshev = 1;
1291  double* buf = NULL;
1292  const double* straightened_input;
1293 
1294  if (chunk == NULL || input == NULL || params == NULL
1295  || params->poly_fit == NULL || params->chebyshev == NULL
1296  || params->inv_chebyshev == NULL)
1297  abort();
1298 
1299  clear_chunk(chunk);
1300 
1301  if (num_of_samples != params->samples_per_chunk)
1302  return PCOMP_STAT_INVALID_BUFFER;
1303 
1304  if (params->period > 0.0) {
1305  buf = malloc(num_of_samples * sizeof(input[0]));
1306  pcomp_straighten(buf, input, num_of_samples, params->period);
1307  straightened_input = buf; /* This preserve const-correctness */
1308  }
1309  else {
1310  straightened_input = input;
1311  }
1312 
1313  if (num_of_samples <= params->poly_fit->num_of_coeffs) {
1314  /* The number of element is so small that is better to
1315  * store
1316  * them uncompressed */

1318  }
1319  else {
1320  double max_residual;
1321 
1322  /* Compute the polynomial fit and the full Chebyshev
1323  * transform */
1324  coeffs
1325  = malloc(sizeof(double) * params->poly_fit->num_of_coeffs);
1326  polyfit_and_chebyshev(params, coeffs, straightened_input,
1327  &max_residual);
1328  apply_chebyshev
1329  = (max_residual >= params->max_allowable_error)
1330  && (params->algorithm != PCOMP_ALG_NO_CHEBYSHEV);
1331 
1332  /* If the Chebyshev transform is needed, truncate it as much
1333  * as possible */
1334  if (apply_chebyshev) {
1335  mask = malloc(pcomp_chunk_cheby_mask_size(num_of_samples));
1336  cheby_coeffs_to_retain = trunc_chebyshev(
1337  params->chebyshev, params->inv_chebyshev,
1338  params->max_allowable_error, mask, max_error);
1339

1341  + params->poly_fit->num_of_coeffs)
1342  < num_of_samples;
1343  if (!chunk->is_compressed) {
1344  free(mask);
1345  mask = NULL;
1346  }
1347  }
1348  else {
1349  /* Assume that num_of_samples > deg(p) + 1 */
1350  chunk->is_compressed
1351  = (max_residual <= params->max_allowable_error);
1352  }
1353  }
1354 
1355  chunk->num_of_samples = num_of_samples;
1356  if (chunk->is_compressed) {
1357  size_t idx;
1358 
1359  chunk->num_of_poly_coeffs = params->poly_fit->num_of_coeffs;
1360  chunk->poly_coeffs = coeffs;
1361  if (apply_chebyshev) {
1362  size_t cheby_idx;
1363 
1364  chunk->num_of_cheby_coeffs = cheby_coeffs_to_retain;
1365  chunk->cheby_mask = mask;
1366  chunk->cheby_coeffs
1367  = malloc(sizeof(double) * cheby_coeffs_to_retain);
1368  cheby_idx = 0;
1369  for (idx = 0; idx < params->chebyshev->num_of_samples;
1370  ++idx) {
1371  if (get_bit(mask, idx)) {
1372  chunk->cheby_coeffs[cheby_idx++]
1373  = params->chebyshev->output[idx];
1374  }
1375  }
1376  }
1377  else {
1378  chunk->num_of_cheby_coeffs = 0;
1379  chunk->cheby_mask = NULL;
1380  chunk->cheby_coeffs = NULL;
1381  }
1382  }
1383  else {
1384  size_t idx;
1385 
1386  if (coeffs != NULL)
1387  free(coeffs);
1388 
1389  chunk->uncompressed = malloc(sizeof(double) * num_of_samples);
1390  for (idx = 0; idx < num_of_samples; ++idx)
1391  chunk->uncompressed[idx] = input[idx];
1392 
1393  if (max_error != NULL)
1394  *max_error = 0.0;
1395  }
1396 
1397  if (buf != NULL) {
1398  free(buf);
1399  }
1400 
1401  return PCOMP_STAT_SUCCESS;
1402 }
